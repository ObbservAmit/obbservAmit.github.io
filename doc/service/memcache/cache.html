from google.appengine.api import memcache
import logging
#

class to manage memcache

class Cache:
#

insert key value pair in memcache

    def insert(self,key,value):
# 
        memcache.add(key=key,value=value,time=120)
#

return value from memcahe

    def get_data(self,key):
# 
        data = memcache.get(key)
        if data is not None:
            return data
        return None
#

add user session value in cache for 3600sec

    def create_user_session(self,email,oauth_token):
# 
        memcache.add(key=email,value=oauth_token,time=3600)
#

checks if session exists

    def check_if_user_session(self,email,oauth_token):
# 
        session = memcache.get(email)
        if session is not None:
            return session
        else:
            return None
