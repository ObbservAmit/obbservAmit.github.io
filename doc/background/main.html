class FetchLocations(webapp2.RequestHandler):
# 
	def post(self):
#

get account key from request

		account_key = ndb.Key(urlsafe=self.request.get('account_key'))
#

get account id from request

		account_id = self.request.get('account_id')
#

get registred_key from request

		registred_key = self.request.get('registred_key')
#

convert urlsafe key into ndb key and get entity

		gbl = ndb.Key(urlsafe=registred_key).get()
#

create fetch object with oauth creds

		fetch = FetchGbl(gbl.creds)
#

fetch location

		results = fetch.get_connected_locations(account_id)
		locations = results.get('locations',None)
#

if location then save it in db

		if locations:
			if len(locations)>0:
#

insert locations in Location modal

				for location in locations:
					location_key = Location().create(account_key,location.get('locationName'),location.get('name'),location.get('primaryPhone'),\
						ndb.GeoPt(location.get('latlng',dict(latitude=00.00)).get('latitude'),location.get('latlng',dict(longitude=00.00)).get('longitude')),\
						location.get('locationKey').get('placeId'),location.get('websiteUrl'))
#

udpate address 

					Address().insert(location_key,location.get('address').get('regionCode'),\
						location.get('address').get('languageCode'),location.get('address').get('postalCode'),\
						location.get('address').get('administrativeArea'),location.get('address').get('locality'),\
						location.get('address').get('addressLines'))
#

update meta data

					Meta().insert(location_key,location.get('metadata').get('mapsUrl'),location.get('metadata').get('newReviewUrl'),\
						location.get('labels',None),location.get('primaryCategory'),location.get('additionalCategories',None))
#

location status

					Status().update_location_status(location_key,location.get('locationState',None))
#

update account location connected flag

				account = account_key.get()
				account.locations_connected = True
				account.put()
#

create and add task schedule task to fetch location insights

				fetch_locations_insights = taskqueue.add(
					url='/fetch/specific_location_insights',
					target='background',
					params={'registred_key':registred_key,\
					'account_key':account_key.urlsafe(),\
					'account_id':account_id
				})

class FetchSpecificLocationInsights(webapp2.RequestHandler):
# 
	def post(self):
		account_key = self.request.get('account_key')
		account_id = self.request.get('account_id')
		registred_key = self.request.get('registred_key')
		gbl = ndb.Key(urlsafe=registred_key).get()
#

init fetch object

		fetch = FetchGbl(gbl.creds)
		for day in xrange(1,60):
			locations_ids = Location().get_location_ids(ndb.Key(urlsafe=account_key))
			if len(locations_ids)>0:
#

fetch insights and save in Insights modal

				for location_id in locations_ids:
					insight = fetch.get_aggregated_insights_last_60_days(location_id,account_id,day)
#

store locations insights in Insights modal

					if insight:
						Insight().insert_latest_insights(insight)
