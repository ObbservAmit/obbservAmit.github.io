from google.appengine.ext import ndb
from role import Role
from registered import Registered
import os
#

ndb modal to manage users

class User(ndb.Model):
# 
    email = ndb.StringProperty()
    brand_name = ndb.StringProperty()
    first_name = ndb.StringProperty()
    last_name = ndb.StringProperty()
    picture_url = ndb.StringProperty(default=None)
    mobile = ndb.IntegerProperty(default=None)
    typ = ndb.StringProperty()
    date= ndb.DateTimeProperty(auto_now_add=True)
    role = ndb.KeyProperty(kind=Role)
    assigned_email = ndb.KeyProperty(kind=Registered)
    connected = ndb.BooleanProperty(default=False)
    demo = ndb.BooleanProperty(default=False)

    def assign_demo_account(self,user_email):
# 
        user = User.query(User.email == user_email).get()
        user.assigned_email = Registered().assign_demo()
        user.connected = False
        user.demo = True
        return user.put()

    def update(self,email,brand_name,first_name,last_name,picture_url,mobile,typ):

        user_key  = User.query(User.email == email).get(keys_only=True)
