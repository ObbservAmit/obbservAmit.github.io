from google.appengine.ext import ndb
from modal.location import Location
#

store listing/location status

class Status(ndb.Model):
# 
	location = ndb.KeyProperty(kind=Location)
	isVerified = ndb.BooleanProperty()
	isGoogleUpdated = ndb.BooleanProperty()
	isPublished = ndb.BooleanProperty()
	canDelete = ndb.BooleanProperty()
	canUpdate = ndb.BooleanProperty()
#

update location status

	def update_location_status(self,location,locationState):
# 
		if locationState:
			status = Status(location=location,isVerified=locationState.get('isVerified',None),\
				isGoogleUpdated=locationState.get('isGoogleUpdated',None),isPublished=locationState.get('isPublished',None),\
				canDelete=locationState.get('canDelete',None),canUpdate = locationState.get('canUpdate',None))
			status.put()
