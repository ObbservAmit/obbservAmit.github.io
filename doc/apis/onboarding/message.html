class EmailResponse(messages.Message):
# 
    (email,date,verified,active,key)

	email = messages.StringField(1)
	date = message_types.DateTimeField(2)
	verified = messages.BooleanField(3)
	active = messages.BooleanField(4)
	key = messages.StringField(5)

return list of EmailResponse objects

class ListResponse(messages.Message):
# 
	emails = messages.MessageField(EmailResponse,1,repeated=True)

class AccountResponse(messages.Message):
# 
    name:string
    account urlsafe key: string
    account id : string
    status : string (verified,unverified,published)
    typ: string
    register_with : string (olv registred email urlsafe key)
    connected : boolean flag (gbl connection status flag)

	name = messages.StringField(1,required=True)
	key = messages.StringField(2)
	account_id = messages.StringField(3,required=True)
	status = messages.StringField(4,required=True)
	typ = messages.StringField(5)
	register_with = messages.StringField(6)
	connected = messages.BooleanField(7)
#

response class return AccountResponse objects list

class AccountListResponse(messages.Message):
