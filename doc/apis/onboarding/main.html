	@endpoints.method(message_types.VoidMessage,SessionResponse,path='/check_if_session_exists',http_method="GET")
	def check_if_session_exists(self,request):
# 
		current_user = endpoints.get_current_user().email()
#

get access token from headers

		oauth_token = request._Message__unrecognized_fields.get(u'access_token')[0]
#

get session from memecache

		session = Cache().check_if_user_session(current_user,oauth_token)
		if session:
			return SessionResponse(session=True)
		else:
			return SessionResponse(session=False)
