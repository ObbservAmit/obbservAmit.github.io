from protorpc import messages
from protorpc import message_types
import endpoints
#

location request message class get location key, start date, end date and account key form clinet

class LocationKey(messages.Message):
# 
	location_key = messages.StringField(1)
	startDate = message_types.DateTimeField(2)
	endDate = message_types.DateTimeField(3)
	account_key = messages.StringField(4)
#

response message class return views_map and views_search

class Views(messages.Message):
# 
	views_map = messages.IntegerField(1,required=True)
	views_search = messages.IntegerField(2,required=True)
#

response class return list of views containing views objects

class ViewsListResponse(messages.Message):
# 
	views = messages.MessageField(Views,1,repeated=True)
#

response class return Action(actions_driving_directions,actions_phone,actions_website)

class Actions(messages.Message):
# 
	actions_driving_directions = messages.IntegerField(1,required=True)
	actions_phone = messages.IntegerField(2,required=True)
	actions_website = messages.IntegerField(3,required=True)
#

response class return Action(actions_driving_directions,actions_phone,actions_website)

class ActionsListResponse(messages.Message):
# 
	actions = messages.MessageField(Actions,1,repeated=True)
#

response class return Searches(indirect_search,direct_search)

class Searches(messages.Message):
# 
	indirect_search = messages.IntegerField(1,required=True)
	direct_search = messages.IntegerField(2,required=True)
#

Location response class return location->(geo_points,location_placeId,name,phone_number,website_url,location_key)

class LocationMessage(messages.Message):
# 
	geo_points = messages.StringField(1,required=True)
	location_placeId = messages.StringField(2,required=True)
	name = messages.StringField(3,required=True)
	phone_number = messages.StringField(4)
	website_url = messages.StringField(5)
	location_key = messages.StringField(6)
#

location response class return location list containing LocationMessage objects

class LocationListResponse(messages.Message):
# 
	locations = messages.MessageField(LocationMessage,1,repeated=True)
#

Address response class return location address details-> (regionCode,languageCode,postalCode,administrativeArea,locality,addressLines,name)

class AddressResponse(messages.Message):
# 
	regionCode = messages.StringField(1,required=True)
	languageCode = messages.StringField(2,required=True)
	postalCode = messages.IntegerField(3,required=True)
	administrativeArea = messages.StringField(4,required=True)
	locality = messages.StringField(5,required=True)
	addressLines = messages.StringField(6,repeated=True)
	name = messages.StringField(7,required=True)
#

MetaResponse response class return location meta data->(labels,mapsUrl)

class MetaResponse(messages.Message):
# 
	labels = messages.StringField(1,repeated=True)
	mapsUrl = messages.StringField(2,required=True)
#

CategorysResponse response class return categorys list location belong to

class CategorysResponse(messages.Message):
# 
	categorys = messages.StringField(1,repeated=True)
#

LocationInsightChange request class get location P1 start date ,P1 end date,P2 start date,P2 end date and account key

class LocationInsightChange(messages.Message):
# 
	p1_start = message_types.DateTimeField(1)
	p1_end = message_types.DateTimeField(2)
	p2_start = message_types.DateTimeField(3)
	p2_end = message_types.DateTimeField(4)
	account_key = messages.StringField(5)
#

location insights change response class return insights change between two periods p1 and p2

class LocationInsightChangeResponse(messages.Message):
# 
    return => (
        name,website_url,location_key,p_change_actions,p1_actions,p2_actions,p_change_views,p1_views,p2_views,p_change_searches
        p1_searches,p2_searches,score,p1_score,p2_score
    )

	name = messages.StringField(1,required=True)
	website_url = messages.StringField(2)
	location_key = messages.StringField(3)
	p_change_actions = messages.StringField(4)
	p1_actions = messages.IntegerField(5)
	p2_actions = messages.IntegerField(6)
	p_change_views = messages.StringField(7)
	p1_views = messages.IntegerField(8)
	p2_views = messages.IntegerField(9)
	p_change_searches = messages.StringField(10)
	p1_searches = messages.IntegerField(11)
	p2_searches = messages.IntegerField(12)
	score = messages.StringField(13)
	p1_score = messages.StringField(14)
	p2_score = messages.StringField(15)
#

Locations response class return list of LocationInsightChangeResponse objects

class LocationInsightChangeResponseList(messages.Message):
# 
	locations = messages.MessageField(LocationInsightChangeResponse,1,repeated=True)
#

response class return graph data which is used to display graph on client side

class ActionInsightGraph(messages.Message):
# 
    dates: list
    actions on phone: list
    actions on website: list
    actions on directions: list

	dates = messages.StringField(2,repeated=True)
	phone = messages.IntegerField(1,repeated=True)
	website = messages.IntegerField(3,repeated=True)
	directions = messages.IntegerField(4,repeated=True)
#

response class return search insights graph data

class SearchInsightGraph(messages.Message):
# 
    dates: list
    searchs direct: list
    searchs indirect: list

	dates = messages.StringField(2,repeated=True)
	direct = messages.IntegerField(1,repeated=True)
	indirect = messages.IntegerField(3,repeated=True)
#

    views search: list
