	@endpoints.method(LocationKey,LocationListResponse, path='/get_locations',http_method="POST")
	def get_locations(self,request):
#

get’s current user

		user = endpoints.get_current_user().email()
#

generate account key from urlsafe accout key

		account_key = ndb.Key(urlsafe=request.account_key)
#

get all location connected with provided accont key

		locations = Location().get_locations([account_key])
		location_list = []
		for location in locations:
			location_list.append(LocationMessage(geo_points=str(location.geo_points),location_placeId=location.location_placeId,\
				name=location.name,phone_number=location.phone_number,website_url=location.website_url,location_key=location.key.urlsafe()))
		return LocationListResponse(locations=location_list)

	@endpoints.method(LocationInsightChange,LocationInsightChangeResponseList, path='/get_location_change_in_insights',http_method="POST")
	def get_location_change_in_insights(self,request):
# 
		user = endpoints.get_current_user().email()
		account_key =  ndb.Key(urlsafe=request.account_key)
		location_keys = Location().get_location_keys([account_key])
		location_list = []
		for location_key in location_keys:
#

get location insights between p2_start and p1_end date p1 period (total_actions,total_views,total_searches)

			p1_insights = Insight().get_insights_sum(Insight().get_multi_insights_from_index(location_key,request.p1_start,request.p1_end))

get location insights between p2_start and p2_end date p2 period (total_actions,total_views,total_searches)

			p2_insights = Insight().get_insights_sum(Insight().get_multi_insights_from_index(location_key,request.p2_start,request.p2_end))
#

append location insight change

			location_list.append(LocationInsightChangeResponse(
				name=location_key.get().name,\
#

return % change in location actions sub(p1_insights,p2_insights)

				p_change_actions='{}%'.format(Insight().calculate_change_in_actions(p1_insights,p2_insights)),\
				p1_actions = p1_insights.get('total_actions'),\
				p2_actions = p2_insights.get('total_actions'),\
#

return % change in location views sub(p1_insights,p2_insights)

				p_change_views='{}%'.format(Insight().calculate_change_in_views(p1_insights,p2_insights)),\
				p1_views = p1_insights.get('total_views'),\
				p2_views = p2_insights.get('total_views'),\
#

return % change in location searches sub(p1_insights,p2_insights)

				p_change_searches='{}%'.format(Insight().calculate_change_in_searches(p1_insights,p2_insights)),\
				p1_searches = p1_insights.get('total_searches'),\
#

chnage in olv score, algo used to generate olv score need to work on

				score = '{0:.2f}'.format(Insight().calculate_score(p1_insights.get('total_actions'),p1_insights.get('total_views'))\
					-Insight().calculate_score(p2_insights.get('total_actions'),p2_insights.get('total_views'))),\
				p1_score = '{0:.2f}'.format(Insight().calculate_score(p1_insights.get('total_actions'),p1_insights.get('total_views'))),\
				p2_score = '{0:.2f}'.format(Insight().calculate_score(p2_insights.get('total_actions'),p2_insights.get('total_views'))),\
				p2_searches = p2_insights.get('total_searches'),\
				website_url=location_key.get().website_url,\
				location_key=location_key.urlsafe()))
		return LocationInsightChangeResponseList(locations=location_list)

	@endpoints.method(LocationKey,LocationInsightChangeResponseList, path='/get_all_location_insights',http_method="POST")
	def get_all_location_insights(self,request):
# 
		user = endpoints.get_current_user().email()
		account_key =  ndb.Key(urlsafe=request.account_key)
		location_keys = Location().get_location_keys([account_key])
		location_list = []
		for location_key in location_keys:
			p1_insights = Insight().get_insights_sum(Insight().get_multi_insights_from_index(location_key,request.startDate,request.endDate))
			location_list.append(LocationInsightChangeResponse(name=location_key.get().name,\
				p_change_actions='{}'.format(p1_insights.get('total_actions')),\
				p_change_views='{}'.format(p1_insights.get('total_views')),\
				p_change_searches='{}'.format(p1_insights.get('total_searches')),\
				score = '{0:.2f}'.format(Insight().calculate_score(p1_insights.get('total_actions'),p1_insights.get('total_views'))),
				website_url=location_key.get().website_url,\
				location_key=location_key.urlsafe()))
		return LocationInsightChangeResponseList(locations=location_list)

return aggregated insights actions

	@endpoints.method(LocationKey,Actions, path='/get_aggregated_actions',http_method="POST")
	def get_aggregated_actions(self,request):
# 
		current_user = endpoints.get_current_user().email()
		account_key = ndb.Key(urlsafe=request.account_key)
		if request.location_key:
#

getting aggregate_insights from cache if present

			aggregate_insights = Cache().get_data('{}#{}#{}#insights'.format(request.location_key,str(request.startDate).split(' ')[0],str(request.endDate).split(' ')[0]))
			if aggregate_insights is None:
				locations = [ndb.Key(urlsafe=request.location_key)]
#

location insights between start amd end date returns future object get results by insights.get_result()

				insights = Insight().get_multi_insights(locations,request.startDate,request.endDate)
#

get aggregated insights

				aggregate_insights = Insight().get_aggregated_insights(insights.get_result())
#

saving insights in memecache

				Cache().insert('{}#{}#{}#insights'.format(request.location_key,str(request.startDate).split(' ')[0],str(request.endDate).split(' ')[0]),aggregate_insights)
		else:
#

aggregated insights for locations connected with provided account_key

			aggregate_insights = Insight().fetch_insight_from_search_index([account_key],request.startDate,request.endDate)
		return Actions(actions_driving_directions=aggregate_insights.get('actions_driving_directions'),\
			actions_phone=aggregate_insights.get('actions_phone'),actions_website=aggregate_insights.get('actions_website'))
